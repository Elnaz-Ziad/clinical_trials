# Columns to keep the same for each grade
id_vars = [
    'meddra_code', 'meddra_soc', 'ctcae_term',
    'definition', 'navigational_note', 'ctcae_v5_change'
]

# Columns that represent different grades
grade_cols = ['grade_1', 'grade_2', 'grade_3', 'grade_4', 'grade_5']

# Unpivot the dataset so that each grade becomes its own row
unpivoted_ctcae_v5 = (
    ctcae_v5.replace({'-': pd.NA})  
             .melt(
                 id_vars=id_vars,
                 value_vars=grade_cols,
                 var_name='grade',
                 value_name='criteria'
             )
             .dropna(subset=['criteria'])  
             .assign(
                 grade_num=lambda d: d['grade']
                 .str.extract(r'(\d+)') 
                 .astype('Int64')
             )
             .sort_values(['meddra_code', 'grade_num'])
             .reset_index(drop=True)
)

In[13]:
# --- Unique code sets ---
v3 = set(v3_to_v4['v3_meddra_code'].dropna().unique())
v4 = set(v4_grade_to_v5_grade['v4_meddra_code'].dropna().unique())
v5 = set(ctcae_v5['meddra_code'].dropna().unique())

# --- Compute total for percentages ---
total = len(v3.union(v4).union(v5))
In[14]:
# --- Create pastel Venn ---
plt.figure(figsize=(6, 6))
venn = venn3(
    subsets=(v3, v4, v5),
    set_labels=('CTCAE v3', 'CTCAE v4', 'CTCAE v5'),
    set_colors=('#9ecae1', '#fdae6b', '#a1d99b'),  # pastel blue, orange, green
    alpha=1  # softer transparency
)

# --- Style the diagram ---
for text in venn.set_labels:
    text.set_fontsize(10)
for text in venn.subset_labels:
    if text:
        count = int(text.get_text())
        pct = 100 * count / total
        text.set_text(f"{count}\n({pct:.1f}%)")
        text.set_fontsize(9)
        text.set_fontweight('normal')  # unbold numbers
        text.set_color('#333333')  # slightly soft gray text

# --- Title and legend ---
plt.title(
    'Overlap of Unique MedDRA Codes Across CTCAE Versions',
    fontsize=14,
    pad=20
)

patches = [
    mpatches.Patch(color='#9ecae1', label='CTCAE v3'),
    mpatches.Patch(color='#fdae6b', label='CTCAE v4'),
    mpatches.Patch(color='#a1d99b', label='CTCAE v5')
]
plt.legend(
    handles=patches,
    loc='lower center',
    bbox_to_anchor=(0.5, -0.08),
    ncol=3,
    frameon=False
)

# Plot a sankey chart

# --- flows (counts) ---
# v3 -> v4 (these sum to N)
v3_in__v4_in   = len(v3 & v4)
v3_in__v4_out  = len(v3 - v4)
v3_out_v4_in   = len(v4 - v3)
v3_out_v4_out  = len(union - (v3 | v4))  # only v5

# v4 -> v5 (these also sum to N and conserve through v4)
v4_in__v5_in   = len(v4 & v5)
v4_in__v5_out  = len(v4 - v5)
v4_out_v5_in   = len(v5 - v4)
v4_out_v5_out  = len(union - (v4 | v5))  # only v3

# --- print counts to verify non-zero links ---
values = [
    v3_in__v4_in,  v3_in__v4_out,  v3_out_v4_in,  v3_out_v4_out,
    v4_in__v5_in,  v4_in__v5_out,  v4_out_v5_in,  v4_out_v5_out
]
print("Link counts (v3v4 then v4v5):", values)
print("Totals:", {"|U|": N, "|v3|": len(v3), "|v4|": len(v4), "|v5|": len(v5)})

if max(values) == 0:
    raise ValueError("All Sankey link values are zero. Check your input sets.")

# --- nodes: order v3, v4, v5 (In / Not in) ---
labels = ["v3: In", "v3: Not in", "v4: In", "v4: Not in", "v5: In", "v5: Not in"]
V3_IN, V3_OUT, V4_IN, V4_OUT, V5_IN, V5_OUT = range(6)

sources = [
    V3_IN,  V3_IN,  V3_OUT, V3_OUT,      # v3 -> v4
    V4_IN,  V4_IN,  V4_OUT, V4_OUT       # v4 -> v5
]
targets = [
    V4_IN,  V4_OUT, V4_IN,  V4_OUT,
    V5_IN,  V5_OUT, V5_IN,  V5_OUT
]

# percentages for hover
perc = [f"{(v / N * 100):.1f}%" for v in values]

# optional node colors
node_colors = ["#6baed6", "#c6dbef", "#fd8d3c", "#fdd0a2", "#74c476", "#c7e9c0"]

fig = go.Figure(data=[go.Sankey(
    arrangement="snap",
    node=dict(
        pad=20, thickness=18,
        label=labels,
        color=node_colors,
        line=dict(color="rgba(0,0,0,0.25)", width=0.5)
    ),
    link=dict(
        source=sources,
        target=targets,
        value=values,
        customdata=perc,
        hovertemplate="<b>%{source.label}</b>  <b>%{target.label}</b><br>"
                      "Count: %{value}<br>"
                      "Percent of union: %{customdata}<extra></extra>",
        color="rgba(128,128,128,0.35)"
    )
)])

fig.update_layout(
    title_text="Sankey of MedDRA Code Membership Across CTCAE v3  v4  v5",
    font=dict(size=12)
)

# --- Display and also save as standalone HTML ---
fig.show()
fig.write_html("/cluster/home/t128190uhn/datasets/ctcae/eda/meddra_sankey.html", include_plotlyjs="cdn")
print("Saved interactive chart to 'meddra_sankey.html' in the working directory.")
